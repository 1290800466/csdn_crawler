1、类和对象的区别与联系：
     1.类是抽象的，概念的，代表一类事物，比如人类，猫类
     2.对象是具体的，实际的，代表一个具体的事物
     3.类是对象的模板，对象是类的一个个体实例

<?php 

//创建一个Cat类
class Cat
{
	public $name;
	public $age;
	public $color;
}


//创建一个对象 cat1
//通过Cat类创建一个cat1对象
$cat1 = new Cat();
$cat1->name = "小白";
$cat1->age = "3";
$cat1->color = "白色";
 
//创建一个对象 cat2
//通过Cat类创建一个cat2对象
$cat2 = new Cat();
$cat2->name = "小花";
$cat2->age = "23";
$cat2->color = "花色";


//如果我们找到一只猫，只要找到$cat1，那么该变量所有相关的属性都通通找到
$catname = "小白";

{
