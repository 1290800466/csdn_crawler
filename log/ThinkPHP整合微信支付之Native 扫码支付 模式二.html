  5.          
  6.         //存储微信的回调
  7.         $xml = $GLOBALS['HTTP_RAW_POST_DATA'];
  8.         $notify->saveData($xml);
  9.          
  10.         //验证签名，并回应微信。
  11.         //对后台通知交互时，如果微信收到商户的应答不是成功或超时，微信认为通知失败，
  12.         //微信会通过一定的策略（如30分钟共8次）定期重新发起通知，
  13.         //尽可能提高通知的成功率，但微信不保证通知最终能成功。
  14.         if($notify->checkSign() == FALSE){
